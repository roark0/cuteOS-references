ENTRY(_Reset)

SECTIONS
{
. = 0x910080;
. = ALIGN(4);

.text : { *(.text) }
.data : { *(.data) }
.bss : { *(.bss) }

. = ALIGN(4);
. = . + 0x1000; /* 4kB of stack memory */
stack_top = .;
}
